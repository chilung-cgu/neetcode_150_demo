class DetectSquares {
    map<pair<int,int>, int> pointCount;
public:
    void add(vector<int> point) {
        pointCount[{point[0], point[1]}]++;
    }
    int count(vector<int> point) {
        int res = 0, x1 = point[0], y1 = point[1];
        for (auto& [p3, cnt] : pointCount) {
            int x3 = p3.first, y3 = p3.second;
            if (abs(x1-x3) == 0 || abs(x1-x3) != abs(y1-y3)) continue; // Check calc diagonal
            res += cnt * pointCount[{x1, y3}] * pointCount[{x3, y1}];
        }
        return res;
    }
};
